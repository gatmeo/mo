\section*{Amazing properties of binomial coefficients --- 2}

{\footnotesize\noindent
``The official theoretical source'' for this set of problems is
Vinberg's article \cite{Vinberg}.
Particularly the following theorems are considered to be known.
%В частности, считаются известными следующие теоремы.

\begin{enumerate}
\item
{\sc Wilson's theorem.} For any prime $p$ (and for primes only)
the equivalence holds $(p-1)!\equiv -1\pmod p$.
\item
{\sc Lukas' theorem.} Write the numbers $n$ and $k$ in base $p$:
\begin{equation}
n=n_dp^d+n_{d-1}p^{d-1}+\ldots+n_1p+n_0,
\qquad
k=k_dp^d+k_{d-1}p^{d-1}+\ldots+k_1p+k_0.
\label{n,k_bin_razlozhenija}
\end{equation}
Then
$
\binom(n,k)\equiv
\binom({n_d},{k_d})\binom({n_{d-1}},{k_{d-1}})\cdot\ldots\cdot
\binom({n_{1}},{k_{1}})\binom({n_{0}},{k_{0}}) \pmod{p}\,.
$

\item
{\sc Kummer's theorem.} The exponent $\ord_p \binom(n,k)$ is equal
to the number of ``carries'' when we add $k$ and $\ell=n-k$ in base~$p$.

\item
{\sc Wolstenholme's theorem.} If $p\geq 5$ then $\binom({2p},p)\equiv 2\pmod{p^3}$, or, that is the same,
$\binom({2p-1},{p-1})\equiv 1\pmod{p^3}$.
\end{enumerate}

\noindent
Remind that $\binom(0,0)=1$, $\binom(n,k)=0$ for $k>n$ and for $k<0$ by definition.

We denote by $p$ a prime number.
For any natural $n$ denote by $(n!)_p$ the product of all integers from 1 to $n$
not divisible by $p$.
If a number $p$ is given the symbols $n_i$, $m_i$ etc. denote the digits of numbers
$n$, $m$ etc.\ in base $p$.
%произведение всех натуральных чисел от 1 до $n$, не делящихся на $p$.
%Если задано число $p$, то символами $n_i$, $m_i$ и т.\,д.\ обозначаются цифры $p$-ичной записи
%чисел $n$, $m$  и т.\,д.

}

For $p=2$ the statement can be easily checked. So we can assume that $p$ is odd prime.
Let $n=x(p-1)+k$. We use induction on $x$.
%Будем доказывать утверждение индукцией по $x$.

The base $x=0$ is trivial: $\binom(k,j)\equiv\binom(k,j)\pmod p$.

To prove the step of induction we need the following property of binomial coefficients:
%Для доказательства перехода воспользуемся свойством биномиальных коэффициентов
$$
\binom(a+b,s)=\sum_i \binom(a,s-i)\binom(b,i)
\qquad\text{(summation in natural bounds)},
$$
both sides of which calculate in how many ways we can choose $s$ balls in the box that contains $a$ black and $b$~white balls.
Let $n=m+(p-1)$. Observe that
$$
\binom(n,{\ell(p\!-\!1)+j})=\binom(m+(p\!-\!1),{\ell(p\!-\!1)+j})
=\sum_{i=0}^{p-1} \binom(m,{\ell(p\!-\!1)+j-i})\binom(p-1,i)
\equiv \sum_{i=0}^{p-1} (-1)^i\binom(m,{\ell(p\!-\!1)+j-i})\!\! \pmod p
$$
(the last equivalence is due to problem \ref{prbl:zadacha_1.1}\,a).
Remark that the sign of the first and last terms in the last sum is ``plus'' .
%sОтметим, что в последней сумме первое и последнее слагаемое присутствуют со знаком плюс.
Now transform the sum from the problem statement:
%Преобразуем теперь интересующую нас сумму.
\begin{multline*}
\sum_\ell \binom(n,{\ell(p-1)+j})
\equiv \\ \equiv
\left( \binom(m,j)- \binom(m,j-1)+\ldots \right)+
\left( \binom(m,{p-1+j})- \binom(m,{p-1+j-1})+\!\ldots\!+ \binom(m,j)\right)
+\\\qquad\qquad\qquad+
\left( \binom(m,{2(p\!-\!1)+j})- \binom(m,{2(p\!-\!1)+j-1})+\!\ldots\!+ \binom(m,{2(p\!-\!1)+j})\right)+\ldots
 \\=
\sum_{i=0}^{m} (-1)^i\binom(m,i) + \sum_\ell \binom(m,{\ell(p\!-\!1)+j}) \pmod p.
\end{multline*}
the first sum is equal to 0, the second sum is equivalent  $\binom(k,j)\pmod p$ by the induction hypothesis.

\solution 3 (algebraical reasoning with Luka's theorem, \cite{Granville-p^q}).
Induction by $n$. Base $n\leq p-1$ is trivial.
Now let $n\geq p$, write all parameters in base $p$, let  $\sigma_p(m)$ denotes the sum of digits of $m$.
It is clear that if  $m\equiv j\pmod p$, then $\sigma_p(m)\equiv j\pmod p$.
The sum under consideration is equal by Luka's theorem to
%Тогда по теореме Люка интересующая нас сумма равна
$$
\sum \binom(n_0,m_0)\binom(n_1,m_1)\dots\binom(n_d,m_d) \pmod p\,,
$$
where the summation is over all $m=\overline{m_d\dots m_1m_0}\leq n$, for which $\sigma_p(m)\equiv j\pmod p$.
This sum is equal to the sum of coefficients of $x^j$, $x^{j+p-1}$, $x^{j+2(p-1)}$, \dots
in the expression
$$
(1+x)^{n_0}(1+x)^{n_1}\dots(1+x)^{n_d}=(1+x)^{\sigma_p(n)}\,.
$$
But it is evident that this sum of coefficients equals
$$
\sum_{\substack{1\leq r\leq \sigma_p(n) \\ r\equiv j \pmod{p-1}} } \binom(\sigma_p(n), r)\,,
$$
which satisfy the induction hypothesis because $1\leq \sigma_p(n)\leq n-1$,
and supply the desired equivalence since  $\sigma_p(n)\equiv n\equiv j\pmod p$.
\end{problem}

\solution 4 (linear algebra, [D]).
The polynomials $x$, $x^2$, \dots, $x^{p-1}$ are linearly independent over $\mathbb{Z}_p$
and form a basis in the space of functions $f\colon\mathbb{Z}_p\to\mathbb{Z}_p, f(0)=0$.
By Fermat's little theorem $(1+x)^n\equiv (1+x)^k \pmod p$.
Applying the relations $x^{i+a(p-1)}\equiv x^i$ to the left hand side,
we obtain that our sum as an element of $\mathbb{Z}_p$ is equal to the coefficient of $x^j$
in the right hand side, i.\,e. $\binom(k,j)$.

%Эта задача Д.Джукича была в 2002 г.\ на олимпиаде 239 школы г.~Санкт-Петербурга,
%а потом засветилась в шорт-листе  IMO-2008.

All the binomial coefficients in the problem statement are odd by Lukas' theorem, therefore,
it is sufficient to check that all the numbers
%Поскольку все биномиальные коэффициенты из условия задачи нечетны
%(по теореме Люка), для доказательства утверждения достаточно проверить, что все числа
$\binom(2^n-1, 1)$, $\binom(2^n-1, 3)$, \dots, $\binom(2^n-1, 2^n-1)$
have distinct reminders modulo $2^n$.
%дают разные остатки при делении на $2^n$.
%Дальше можно  действовать по-разному.

\medskip
\solutionn 1 ([D]).  Assume by the contrary that % Предположим противное, пусть
$\binom(2^n-1, k)\equiv \binom(2^n-1, m) \pmod{2^n}$ for odd $k$ and $m$, $k>m$.
Observe that %Заметим, что
\begin{multline*}
\binom(2^n-1, k)=\binom(2^n, k)-\binom(2^n-1, k-1)=\binom(2^n, k)-\binom(2^n, k-1)+\binom(2^n-1, k-2)
=\dots
=\\=
\binom(2^n, k)-\binom(2^n, k-1)+\binom(2^n, k-2)-\ldots-\binom(2^n, m+1)+\binom(2^n-1, m)\,.
\end{multline*}
In particular
$$
\binom(2^n, k)-\binom(2^n, k-1)+\binom(2^n, k-2)-\ldots-\binom(2^n, m+1)\equiv 0\pmod{2^n}\,.
$$
Calculate the exponent $\ord_2 \binom(2^n,r)$ by Kummer's theorem.
%Теорема Куммера позволяет для каждого $r$ легко вычислить показатель $\ord_2 \binom(2^n,r)$, а именно, если
If $\ord_2 r=a$ then we have $n-a$ carries in addition $r$ and $2^n-r$
(it is clear by the standard algorithm of addition),
%, то при сложении $r$ и $2^n-r$ произойдет $n-a$ переносов (это очевидно из алгоритма сложения столбиком),
%и значит,
hence $\ord_2 \binom(2^n,r)=n-a$.
In particular  $2^n\mid\binom(2^n, r)$  for odd $r$, that allows us
to consider only one half of summands:
%что позволяет отбросить в последнем сравнении половину слагаемых:
$$
\binom(2^n, k-1)+\binom(2^n, k-3)+\ldots+\binom(2^n, m+1)\equiv 0\pmod{2^n}\,.
$$
Now all the  $\binom(2^n,i)$ in the left hand side have even parameter $i$,
%Другое следствие из приведенных рассуждений состоит в том, что у всех слагаемых $\binom(2^n,i)$
%в левой части параметр $i$ четный и поэтому
therefore $\ord_2 \binom(2^n,x)<n$.

We will prove that this congruence is impossible and obtain a contradiction.
%Докажем теперь, что выполнение этого сравнения невозможно.
Choose $x$ with minimal  $\ord_2 \binom(2^n,x)$.
Since ${\ord_2 \binom(2^n,x)<n}$ and the whole sum is divisible by $2^n$, there exists $y$,
for which $\ord_2 \binom(2^n,x)=\ord_2 \binom(2^n,y)$.
Then the binary representations of $x$ and $y$ end with equal number of 0's,
and hence there exists $z$ between $x$ and $y$ which binary representation ends with bigger number of 0's.
%оканчиваются на одинаковое число нулей, поэтому между $x$ и $y$ найдется число~$z$, оканчивающееся на большее число нулей.
Then $\ord_2 \binom(2^n,z)<\ord_2 \binom(2^n,x)$, a contradiction.

\solutionn 2 ([CSTTVZ]). Induction by $n$. We prove the step of induction.
Let the statement be proven for all numbers less than $n$.
%Кроме того, мы будем вести рассуждения по индукции, считая, что для меньших значений $n$ утверждение задачи уже доказано.
Assume by the contrary that there exist $k$ and $\ell$,
$k\ne \ell$, $0\leq k,\ell\leq 2^n-1$, such that $\binom(2^n-1, 2k+1)\equiv \binom(2^n-1, 2\ell+1)\mod{2^n}$.
Observe that
\begin{align}
\binom(2^n-1,2k+1)&=
\biggl(\frac{2^n}1-1\biggr)\biggl(\frac{2^n}2-1\biggr)\dots\biggl(\frac{2^n}{2k+1}-1\biggr)
=\notag \\&=
\biggl(\frac{2^n}1-1\biggr)\biggl(\frac{2^n}3-1\biggr)\dots\biggl(\frac{2^n}{2k+1}-1\biggr)\cdot
\biggl(\frac{2^{n-1}}1-1\biggr)\biggl(\frac{2^{n-1}}2-1\biggr)\dots\biggl(\frac{2^{n-1}}{k}-1\biggr)
= \label{egn:serb_reduction}\\&=
\biggl(\frac{2^n}1-1\biggr)\biggl(\frac{2^n}3-1\biggr)\dots\biggl(\frac{2^n}{2k+1}-1\biggr)\cdot
\binom(2^{n-1}-1,k)
\equiv\notag \\&\equiv
(-1)^{k+1}\binom(2^{n-1}-1,k) \pmod{2^n}
\notag
\end{align}
and analogously $\binom(2^n-1,2\ell+1)\equiv (-1)^{\ell+1}\binom(2^{n-1}-1,\ell) \pmod{2^n}$.
It follows by induction hypothesis that both $k$ and $\ell$ can not be odd.
%По индукционному предположению, отсюда следует, что $k$ и $\ell$ не могут быть оба нечетными.
Besides, due to the symmetry $\binom(2^n-1,r)=\binom(2^n-1,2^n-1-r)$
the problem statement means that all the ``even'' binomial coefficients $\binom(2^n-1,2r)$
are pairwise distinct modulo $2^n$ and form the same set of residues as ``odd'' binomial coefficients  $\binom(2^n-1,2r+1)$.
%утверждение задачи означает также, что все биномиальные коэффициенты с четными показателями --- $\binom(2^n-1,2r)$ --- тоже попарно различны и по модулю $2^n$ образуют то же множество, что и биномиальные коэффициенты с нечетными показателями.
Therefore $k$ and $\ell$ can not be even simultaneously.

It remains to consider a case when $k$ and $\ell$ have distinct parity,
%Осталось разобрать случай, когда $k$ и $\ell$ разной четности, пусть
say $k=2a+1$, $\ell=2b$. Then
$$
\binom(2^{n-1}-1,2a+1)+\binom(2^{n-1}-1,2b)\equiv 0\pmod{2^n}\,.
$$
If $a=b$ the congruence is impossible because $\binom(2^{n-1}-1,2a)$ is odd and
$$
\binom(2^{n-1}-1,2a+1)+\binom(2^{n-1}-1,2a)
=
\binom(2^{n-1}-1,2a)\biggl( 1+ \frac{2^{n-1}-1-2a}{2a+1}\biggr)
=
\binom(2^{n-1}-1,2a)\cdot\frac{2^{n-1}}{2a+1}\equiv 2^{n-1}\pmod{2^n}\,.
$$
If $b\ne a$, then $\binom(2^{n-1}-1,2a)\ne \binom(2^{n-1}-1,2b)$ by the induction hypothesis,
Since $\binom(2^{n-1}-1,2a)+\binom(2^{n-1}-1,2a+1)$ is divisible by $2^{n-1}$,
the sum $\binom(2^{n-1}-1,2b)+\binom(2^{n-1}-1,2a+1)$ can not be divisible by $2^{n-1}$.
\end{problem}

\solution 1 (without Kummer's theorem).
This is solution from the excellent book \cite{spb}.
Assume that all these numbers are divisible by $m$. Then the numbers
%Допустим, что все эти числа делятся на $m$. Тогда числа
\begin{align*}
 \binom(n+k-1,k-1)&= \binom(n+k,k)- \binom(n+k-1,k)\;,\\
 \binom(n+k-2,k-1)&= \binom(n+k-1,k)- \binom(n+k-2,k)\;,\\
&\ldots\\
 \binom(n,k-1)&= \binom(n+1,k)- \binom(n,k)
\end{align*}
are also divisible by $m$. Then analogously $m$ divides all the numbers
$ \binom(n+i,j)$, where $i\le j$ are arbitrary nonnegative integers.
But $ \binom(n,0)$ ($i=j=0$) is not divisible by $m$. A contradiction.

\medskip
\solution 2 (Kummer's theorem).
Let $p$ be a prime divisor of $m$. Prove that at least one of the numbers
$\binom(n,k)$, $\binom({n+1}, k)$, \dots, $\binom({n+k},k)$ is not divisible by $p$.
%Запишем $k$ в системе счисления по основанию $p$.
By Kummer's theorem if we choose  $\ell$  ($n-k\leq \ell\leq n$) such that
the addition $k+\ell$ fulfills in base $p$ without carries
%в системе счисления по основанию $p$ выполнялось без переносов,
then the binomial coefficient $\binom(k+\ell,k)$ is not divisible by $p$.

We will explain how to choose $\ell$ by giving a concrete example.
%Это сделать совсем нетрудно. Мы ограничимся рассуждением на конкретном примере.
Let $p=7$, $k=133$. We will write all the numbers in base 7. % (здесь и далее числа записаны в семиричной системе счисления).
Since we try to choose $\ell$ in the set of $k+1$ numbers,
%Поскольку диапазон, в котором мы ищем число $\ell$, содержит $k+1$ число,
we can always choose $\ell$ such that $k+\ell$ to be one of the following numbers
$$
\dots133,  \quad \dots 233, \quad,  \dots, \quad \dots633.
$$
(Remind that 6 is the greatest digit in our example.) It is clear that the addition
$k+\ell$ fulfills without carries.

\medskip
b)
%Докажите, что для любого $k$ найдется бесконечно много таких $n$, что все числа
%$\binom(n,k)$, $\binom({n+1},k)$, \dots, $\binom({n+k-1},k)$ делятся на $m$.
We found this problem in \cite{Gashkov}.
It is not difficult to construct $n$ by Kummer's theorem. Let $\ord_p m =s$,
and $k$ have $d+1$ digits in base $p$.
Let $n\kratno p^{d+s+1}$. Then the representations of numbers $n-k$, $n-k+1$, \dots, $n-1$
contain digits $(p-1)$ in positions from $(d+2)$ to $(d+s+2)$.
When we add $k$ to these numbers we have carries in these positions.
Therefore by Kummer's theorem all the corresponding binomial coefficients are divisible by  $p^s$.

Since it is not difficult to combine our reasoning for distinct $p$, the statemetn is proven.
%Поскольку условия, наложенные на $n$, легко совмещаются для разных $p$, мы получаем отсюда требуемое.

\begin{problem}
This is an exercise on reading an article. The statement is proven in article \cite{Vinberg}.
%Утверждение доказано в статье Винберга, но доказательство не выделено явно.
Observe that
$$
2\sum_{i=1}^{p-1}\frac1i =\sum_{i=1}^{p-1}\frac1i+\frac1{p-i} =p\sum_{i=1}^{p-1}\frac1{i(p-i)}\,.
$$
Hence the sum under consideration is divisible by $p$.
Since $\frac1i \equiv -\frac1{p-i}\pmod p$, it remains to check that
$$
\smash{\sum_{i=1}^{p-1}\frac1{i^2} \equiv 0\pmod{p}.}
$$
But $\frac1{1^2}$, $\frac1{2^2}$, \dots, $\frac1{(p-1)^2}$ modulo $p$
is the same set as\footnote{ \ These sets coincide because they contain $p-1$ element each,
and it is clear that all the reminders in each set are non zero and pairwise distinct.}, что
$1^2$, $2^2$, \dots, $(p-1)^2$. Therefore it is sufficient to prove that
\begin{equation}
\sum_{i=1}^{p-1} i^2 \equiv 0\pmod{p}.
\label{eq:SumSquaresModP}
\end{equation}
Let $\sum\limits_{i=1}^{p-1} i^2 \equiv s\pmod{p}$. It $p>5$ we can always choose $a$, such that $a^2\not\equiv 1\pmod p$.
Then the sets $\{1, 2, \dots, p-1\}$ and $\{a, 2a, \dots, (p-1)a\}$ coincide (the proof is the same as in the footnote) and
$$
s\equiv \sum_{i=1}^{p-1} i^2 = \sum_{i=1}^{p-1} (ai)^2 =a^2 \sum_{i=1}^{p-1} i^2 \equiv a^2s \pmod p\,.
$$
Thus $s\equiv 0\pmod p$.
\end{problem}

\prod_{i=0}^{p-1}(x-1-i^2).
$$
Change the variable $x-1=t^2$ and obtain
$$

(ij)^2+i^2+j^2+1\equiv i^2+j^2+2 \pmod p.
$$
Therefore,
$$

=\frac{1}{i^2+1}+\frac{1}{j^2+1}\pmod p.
$$

\solution 2 ([D]). Prove that $\binom(kp, {mp})\equiv \binom(k,m)\pmod{p^2}$ by induction on $m$.

To prove the base $m=1$ we have to check that $\binom(pk,p)-\binom(k,1)\equiv 0\pmod{p^2}$. We have
\begin{equation}
\binom(pk,p)-\binom(k,1)=\frac{pk(pk-1)\dots(pk-p+1)}{p!}-k=
\biggl(\frac{(pk-1)(pk-1)\dots(pk-p+1)}{(p-1)!} -1\biggr).
\label{eqn:didin1}
\end{equation}
Split the multipliers in the numerator onto pairs:
$$
(pk-i)(pk-p+i)\equiv pi^2-i^2 \pmod{p^2}.
$$
We see that the product modulo~$p^2$ of each pair does not depend on $k$.
%Как видим, по модулю $p^2$ произведение чисел в парах не зависит от $k$.
Therefore the difference \eqref{eqn:didin1} modulo~$p^2$ does not depend on $k$, too.
Since it is equal to 0 for $k=1$, it is equal to 0 for all $k$.

The step of induction. Let $\binom(kp, {(m-1)p})\equiv \binom(k,m-1)\pmod{p^2}$. We have
\begin{multline}
\binom(kp,mp)= \binom(kp, {(m-1)p})\cdot \frac{(p(k-m)+1)(p(k-m)+1)\dots(p(k-m)+p)}{pm(pm-1)\dots(pm-p+1)}
=\\=
\binom(kp, {(m-1)p})\cdot \frac{(p(k-m)+1)(p(k-m)+1)\dots(p(k-m)+p-1)}{(pm-1)\dots(pm-p+1)}
\cdot\frac{k-m+1}{m}.
\label{eqn:didin2}
\end{multline}
Remark that both fractions are correctly defined modulo $p^2$. As in the proof of base,
the expression in the numerator of big fraction does not depend (modulo~$p^2$) on~$k$.
Then we can put $k=0$ for the calculating the fraction modulo $p^2$
and obtain that it is congruent to 0.
For the remaining part of the expression we can apply the induction hypothesis and obtain
%Пользуясь этим соображением и предположением индукции, мы можем заменить правую часть \eqref{eqn:didin2} на
$$
\equiv \binom(k, {m-1})\cdot\frac{k-m+1}{m}=\binom(k,m)\pmod{p^2}.
$$

\frac{(2n\!+\!1)\cdot 2n\dots(n\!+\!2)}{n!}\cos x.
$$
Now integrate it\footnote{ \ When we were learning the rules of multiplication, we just memorized that ``minus by minus equals plus''. In this formula we multiply signs. If we need to multiply $n$ minuses, the record $(-)^n$ seems to be appropriate. So we leave the old-fashioned notation $(-)^n$, used by the author, instead of the modern one $(-1)^n$.
}
over the interval $[0,\frac\pi2]$:
\begin{align*}
2^{2n}\int\cos^{2n+1}x\,dx &= \frac{\sin(2n+1)x}{2n+1}+\frac{2n+1}{2n-1}\sin(2n-1)x+\ldots,
\\
2^{2n}\int\limits_0^{\pi/2}\cos^{2n+1}x\,dx &= (-)^n\biggl(\frac{1}{2n+1}-\frac{2n+1}{2n-1}+\ldots\biggr).
\end{align*}
Every first grade student of university knows that it is convenient to use integration by parts for calculating this integral:
%Но любой первокурсник знает, что куда проще этот интеграл вычисляется с помощью формулы понижения, для получения которой нужно всего лишь проинтегрировать по частям:
\begin{multline*}
I_{2n+1}=\int\limits_0^{\pi/2}\cos^{2n+1}x\,dx = \int\limits_0^{\pi/2}\cos^{2n}x\cos x\,dx
= \cos^{2n}x\sin x\biggm\vert_{0}^{\pi/2}+2n\int\limits_0^{\pi/2}\cos^{2n-1}x\sin^2 x\,dx
=\\=0+2n\int\limits_0^{\pi/2}\cos^{2n-1}x(1-\cos^2 x)\,dx=2n\cdot I_{2n-1}-2n\cdot I_{2n+1}\,,
\end{multline*}
therefore $I_{2n+1}=\frac{2n}{2n+1}\cdot I_{2n-1}$. Since $I_1=1$, we can apply the formula $n$ times and obtain
$$
\int\limits_0^{\pi/2}\cos^{2n+1}x\,dx=
\frac{2n\cdot(2n-2)\dots2}{(2n+1)(2n-1)\dots 3}\,.
$$
Equating of these two results give us the formula
$$
2^{2n}\frac{2n\cdot(2n-2)\dots2}{(2n+1)(2n-1)\dots 3}=
(-)^n\biggl(\frac{1}{2n+1}-\frac{2n+1}{2n-1}+\ldots+\frac{(2n\!+\!1)\cdot 2n\dots(n\!+\!2)}{n!}\biggr).
$$
Let $p=2n+1$ be a prime number. We obtain the desired congruence by multiplying the last formula by~$p$:
%--- простое число, то домножая на $p$, мы сразу получаем требуемое сравнение
$$
2^{2n}\frac{2n\cdot(2n-2)\dots2}{(2n-1)(2n-3)\dots 3}\equiv (-)^n \pmod{p^2}\,.
$$

\medskip
\solutionn 2 ([CSTTVZ]). We will use the following notations:
$$
A=\sum_{i=1}^{\frac{p-1}2}\frac1i, \qquad
B=\sum_{1\leq i<j\leq\frac{p-1}2}\frac1{ij}, \qquad
C=\sum_{\substack{1\leq i \leq p-1\\ i \text{ is odd}}}\frac1i.
$$
%Так как $A\equiv 0 \pmod p$,  то
Then $A^2=\smash{\sum\limits_{i=1}^{\frac{p-1}2}\frac1{i^2}+2B}\equiv 2B\pmod p$ by the problem 4.3b).
So $A^2\equiv 2B\pmod p$. Further,
$$
2C+A=
\sum\limits_{\substack{1\leq i \leq p-1\\ i \text{ нечетно}}}\frac2i+
\sum_{i=1}^{\frac{p-1}2}\frac2{2i} =
\sum_{i=1}^{p-1}\frac2i\equiv0\pmod{p^2}.
$$
So $C\equiv-\frac12 A\pmod{p^2}$.

Now transform modulo $p^3$ the parts of the given congruence.
The l.h.s. is
$$
(-1)^\frac{p-1}2\binom(p-1,\frac{p-1}2)
\equiv
\Bigl(1-\frac p1\Bigr)\Bigl(1-\frac p2\Bigr)\dots\Bigl(1-\frac p{\frac{p-1}2}\Bigr)
\equiv
1-pA+p^2B \equiv 1-pA+\frac12 p^2A^2\pmod{p^3}.
$$
For transforming the r.h.d observe that
\begin{multline*}
2^{p-1}=\frac{2\cdot 4\cdots(p-1)}{1\cdot2\cdots\frac{p-1}2}\cdot\frac{(p+1)\cdots(2p-2)}{\frac{p+1}2\cdots(p-1)}
=\frac{(p+1)\cdots(2p-2)}{1\cdot3\cdot5\cdots(p-2)}
=\\=
\Bigl(\frac p1+1\Bigr)\Bigl(\frac p3+1\Bigr)\dots\Bigl(\frac p{p-1}+1\Bigr)
\equiv 1+pC+\frac12p^2C^2\equiv 1-\frac12 pA+\frac18p^2A^2\pmod{p^3}\,.
\end{multline*}
Then we have
$$
4^{p-1}\equiv \Bigl(1-\frac12 pA+\frac18p^2A^2\Bigr)^2\equiv 1-pA +\frac14p^2A^2+2\cdot\frac18P^2A^2
= 1-pA+\frac12 p^2A^2\pmod{p^3}.
$$
So the l.h.s. is congruent to the r.h.s.
\end{problem}

\begin{problem}
%Пусть $p$ и $q$ --- различные простые числа. Докажите, что сравнение
%$\binom(2pq-1,pq-1)\equiv 1\pmod{pq}$ выполнено в том и только в том случае, когда
%$\binom(2p-1,p-1)\equiv 1\pmod{q}$ и $\binom(2q-1,q-1)\equiv 1\pmod{p}$.
We found this statement in \cite{Xin}.
Since $2pq-1=(2q-1)p+p-1$, the last digit of the number $2pq-1$ in base $p$ is $p-1$,
and the remaining digits form the number $2q-1$. Similarly the last digit of the number $pq-1$ in base $p$
is $p-1$, and the remaining part forms the number $q-1$.
By Lukas' theorem $\binom(2pq-1,pq-1)\equiv \binom(2q-1,q-1)\binom(p-1,p-1)\equiv \binom(2q-1,q-1)\pmod p$.
On the other hand since $\binom(2pq-1,pq-1)\equiv1\pmod{pq}$, then $\binom(2pq-1,pq-1)\equiv1\pmod{p}$.
So $\binom(2q-1,q-1)\equiv 1\pmod{p}$. Analogously $\binom(2p-1,p-1)\equiv 1\pmod {q}$.

The inverse statement is trivial.
\end{problem}

\subsection{Sums of binomial coefficients}


\begin{problem}
a) %Докажите, что $\sum\limits_{k=0}^{3^a-1} C_{2k}^k$ делится на 3.
It follows from problem 1.3.
If $\Delta_0^0$ is a triangle consisting of the first 3 rows of the 3-arithmetical Pascal triangle,
then the sum of its central binomial coefficients is divisible by 3. For arbitrary $a$
the sum under consideration contains elements of several central triangles, which are multiples of $\Delta_0^0$.
So the total sum is divisible by 3, too.

Another solution ([CSTTVZ]) we can derive from the identity
$\binom(2k,k)=\smash[b]{\sum\limits_{i=0}^k{\binom(k,i)}^2}$. Then
$\sum\limits_{k=0}^{3^a-1} C_{2k}^k=\sum\limits_{k=0}^{3^a-1}\sum\limits_{i=0}^k{\binom(k,i)}^2$.
Since $1^2=2^2=1$, $0^2=0 \pmod 3$, the last sum modulo 3 equals the number of nonzero elements
in the first $3^a$ rows of the Pascal triangle. This number is calculated in the problem 2.1a),
it is divisible by 3.

\medbreak
b) %Докажите, что $\sum\limits_{k=0}^{3^a-1} C_{2k}^k$ делится на $3^a$.
Solution of [D]. The sum is a coefficient of $x^{3^a-1}$ in the polynomial
\begin{multline*}
x^{3^a-1}\biggl( 1+ \frac{(x+1)^2}{x}+\frac{(x+1)^4}{x^2}+\ldots+\frac{(x+1)^{2(3^a-1)}}{x^{3^a-1}}  \biggr)
=
\frac{\frac{(x+1)^{2\cdot 3^a}}{x^{3^a}}-1}{\frac{(x+1)^2}{x}-1}\cdot x^{3^a-1}=
\frac{(x+1)^{2\cdot 3^a}-x^{3^a}}{x^2+x+1}
=\\[3pt]=
\frac{x^{2\cdot 3^a}+\binom(2\cdot 3^a,1)\cdot x^{2\cdot 3^a-1}+\binom(2\cdot 3^a,2)\cdot x^{2\cdot 3^a-2}+\ldots+1 - x^{3^a}}{x^3-1}\cdot(x-1)\,.
\end{multline*}
In order to find this coefficient we will perform the long division of the numerator by the denominator
and then multiply the result by  $(x-1)$.
%Чтобы найти нужный коэффициент, достаточно поделить числитель на знаменатель ``в столбик'',
%и~потом домножить результат на $(x-1)$.
We do not need to find the quotient at whole, it is sufficient to perform the division till
the moment when the coefficient of  $x^{3^a-2}$ will be found, remind that we are trying to find this coefficient modulo $3^a$ only.
Since for $b\nekratno 3$ all the binomial coefficient $\binom(2\cdot 3^a,b)$ are divisible by  $3^a$ (by Kummer's theorem),
we can collect all these coefficient in a separate sum. When we divide this sum by $x^3-1$ all the coefficients of
the quotient are divisible by $3^a$ therefore we can discard this sum.
%Таким образом, не нужно даже доводить деление до конца,  достаточно довести его до нахождения коэффициента при $x^{3^a-2}$, кроме того, напомним, результат нас интересует лишь по модулю $3^a$. Отметим, что при $b\nekratno 3$ все биномиальные коэффициенты
%$\binom(2\cdot 3^a,b)$ делятся на $3^a$  по теореме Куммера.
%Сгруппируем слагаемые с этими коэффициентами и будем делить их сумму на $x^3-1$ отдельно. Очевидно,
%все коэффициенты частного будут тоже делиться на~$3^a$, поэтому все эти слагаемые можно отбросить.
The remaining expression is
$$
\frac{x^{2\cdot 3^a}+\binom(2\cdot 3^a,3)\cdot x^{2\cdot 3^a-3}+\binom(2\cdot 3^a,6)\cdot x^{2\cdot 3^a-6}+\ldots+1 - x^{3^a}}{x^3-1}\cdot(x-1).
$$
All the exponents in the numerator are divisible by 3, hence after division by $x^3-1$ all the exponents of the quotient are divisible by 3, too, and after the multiplying it by $x-1$, there will be no exponents of the form $3k+2$.
So the coefficient that we seek equals $0\pmod{3^a}$.

\end{problem}

\begin{problem}
%Пусть  $C_k=\frac1{k+1}C_{2k}^k$ --- последовательность чисел Каталана.
%Докажите, что $\sum\limits_{k=1}^n C_k \equiv1 \pmod 3$ тогда и только тогда, когда троичное разложение
%числа $n+1$ содержит хотя бы одну цифру 2.
This problem was published in Monthly \cite{Chapman}.
Since
$$
\binom(2n+2,n+1)-4\binom(2n,n)=2\cdot\frac{2n+1}{n+1}\binom(2n,n)-4\binom(2n,n)=-2C_n\,,
$$
then $C_n\equiv \binom(2n+2,n+1)-\binom(2n,n)\pmod 3$. Therefore this sum is telescopic modulo 3:
$$
\sum_{k=1}^n C_k\equiv
\left(\binom(2n+2,n+1)-\binom(2n,n)\right)+\left(\binom(2n,n)-\binom(2n-2,n-1)+\ldots\right)=
\binom(2n+2,n+1)+1\pmod 3.
$$
So by Kummer's theorem we have to clarify  when we have at least one carry in
the addition of the number ${(n+1)}$ with itself in base 3.
It it clear that it happens only if $n+1$ contains at least one 2 in base 2.
%в троичной системе счисления приводит к появлению хотя бы одного переноса.
%Очевидно, это может быть в том и только том случае, когда в записи $n+1$ есть хотя бы одна двойка.
\end{problem}

\begin{problem}
%Пусть $p\geq 3$, $k=[2p/3]$.
%Докажите, что сумма $\binom(p,1) +\binom(p,2) +\ldots+\binom(p,k)$ делится на $p^2$.
This is problem A5 of Putnam Math. Competition, 1998.
Since $\frac1p\binom(p,n) \equiv \frac{(-1)^{n-1}}n\pmod p$, we have
$$
\sum_{n=1}^k\frac1p\binom(p,n)\equiv
\sum_{n=1}^k  \frac{(-1)^{n-1}}n=
\sum_{n=1}^k  \frac{1}n- 2 \sum_{n=1}^{[k/2]}  \frac{1}{2n} \equiv
\sum_{n=1}^k  \frac{1}n+ \sum_{n=p-[\frac{k}2]}^{p-1}  \frac{1}n
\overset{*}=
\sum_{n=1}^{p-1}  \frac{1}n \equiv 0 \pmod p.
$$
The summation in the sum to the left of asterisk really starts from $n=k+1$
(it is easy to check: for $p=6r+1$  we have $k=4r$ and $p-[\frac{k}2]=4r+1=k+1$, similarly for $p=6r+5$).
\end{problem}


\begin{problem}
This statement is from \cite{Chapman-2}.
Solution [CSTTVZ]. Induction on $n$. The base is trivial. Prove the induction step from $n'=n-(p-1)$ to $n$.
Let $q=\frac{n}{p-1}$. Since
$$
\binom(n'+p-1,{x(p-1)}) = \sum_{i=0}^{p-1} \binom(p-1, i)\binom(n',{x(p-1)-i})\,,
$$
we can rewrite the sum under consideration in the form
\begin{multline}
\binom(n, p\!-\!1)+\binom(n, {2(p\!-\!1)})+\binom(n, {3(p\!-\!1)})+\ldots
=\sum_{x=1}^q\sum_{i=0}^{p-1} \binom(p\!-\!1, i)\binom(n',{x(p\!-\!1)-i})
=\\=
\sum_{i=0}^{p-1} \left(\binom(p\!-\!1, i) \sum_{x=1}^q\binom(n',{x(p\!-\!1)-i})\right).
\label{eq:sol5.4}
\end{multline}
By the problem 1.1\,a) we have $\binom(p-1,i)\equiv(-1)^i\pmod{p}$; let $\binom(p-1,i)=ap+(-1)^i$.
By the problem 1.6 we have $\smash[b]{\sum\limits_{x=1}^q}\binom(n',{x(p-1)-i})\equiv \binom(p-1,i)\equiv(-1)^i\pmod{p}$
 for $i=0$, 1, \dots,~$p\!-\!2$;  let $\smash[b]{\sum\limits_{x=1}^q}\binom(n',{x(p-1)-i})=bp+(-1)^i$.
Then
\begin{multline*}
\binom(p\!-\!1, i) \smash{\sum_{x=1}^q}\binom(n',{x(p\!-\!1)-i}) = \bigl(ap+(-1)^i\bigr)\bigl(bp+(-1)^i\bigr)
\equiv 1+(-1)^i(ap+bp)
= \\ =
1+ (-1)^i\left(\!\binom(p\!-\!1,i)\!+\!\sum\limits_{x=1}^q\binom(n',{x(p\!-\!1)\!-\!i})-2\cdot(-1)^i\!\right)
=(-1)^i\left(\!\binom(p\!-\!1,i)\!+\!\sum\limits_{x=1}^q\binom(n',{x(p\!-\!1)\!-\!i})\!\right)-1 \!\!\!\pmod{p^2}.
\end{multline*}
Remind that these transformations hold for $0\leq i\leq p-2$. We can continue equality \eqref{eq:sol5.4},
by separating the summand for $i=p-1$:
\begin{multline*}
\sum_{i=0}^{p-1} \left(\binom(p\!-\!1, i) \sum_{x=1}^q\binom(n',{x(p\!-\!1)\!-\!i})\right)\equiv
\sum_{i=0}^{p-2} \left((-1)^i\left(\binom(p\!-\!1,i)+\sum\limits_{x=1}^q\binom(n',{x(p\!-\!1)\!-\!i})\right)-1\right)+
\sum\limits_{x=0}^{q-1}\binom(n',{x(p\!-\!1)})
=\\ =
\sum_{i=0}^{p-2}(-1)^i\binom(p\!-\!1,i) +
\sum_{i=0}^{p-2} \left((-1)^i\sum\limits_{x=1}^q\binom(n',{x(p\!-\!1)-i})\right)-(p-1)+
\binom(n',0)+\sum\limits_{x=1}^{q-1}\binom(n',{x(p\!-\!1)}).
\end{multline*}
The first sum here equals $-1$, because $\binom(p-1,0)-\binom(p-1,1)+\binom(p-1,2)+\ldots=0$.
By the same reasons the second (double) sum together with the summand $\binom(n',0)$ equals 0.
The last sum equals $1+p(n'+1)$ by the induction hypothesis. Therefore the whole expression equals
$-1+0-p+1+1+p(n'+1)=1+pn'$. This is exactly what we need because $1+p(n+1)=1+p(n'+p-1+1)\equiv 1+pn'\pmod{p^2}$.
\end{problem}


\begin{problem}
%Докажите, что при $0\leq j\leq p-1<n$ и $q=[\frac{n-1}{p-1}]$
%$$
%\sum_{m: m\equiv j \pmod{p}} (-1)^m\binom(n,
%m)\equiv 0\pmod{p^q}.
%$$
This is result of Fleck, 1913, it is cited in \cite{Granville-p^q}.
Solution [CSTTVZ].

For $p=2$ the sum is not alternating and the result is trivial.
Let $p$ be odd. We use the induction on~$q$. The base follows from the statement 2.5\,a).
Prove the induction step from $n'=n-(p\!-\!1)$ to~$n$.
The expression $\sum_x$ below denotes the summation over $x$ in natural bounds
(i.e.~in bounds for which all the binomial coefficients are correctly defined). We have
\begin{multline*}
\pm\!\!\!\!\!\!\!\sum_{m: m\equiv j\!\!\! \pmod{p}} (-1)^m\binom(n,m)
=\sum_{x}(-1)^x\binom(n'+p-1,xp+j)
=\sum_{x}(-1)^x\sum_{i=0}^{p-1}\binom(p-1, i)\binom(n',xp+j-i)
=\\=
\sum_{i=0}^{p-1}\binom(p-1, i) \sum_{x}(-1)^x \binom(n',xp+j-i).
\end{multline*}
By the induction hypothesis $\smash{p^{q-1} \bigm| \sum\limits_{x}(-1)^x \binom(n',xp+j-i)}$;
by the problem 1.1\,a)  $\binom(p-1,i)\equiv(-1)^i\!\pmod{p}$. Therefore
$$
\sum_{i=0}^{p-1}\binom(p-1, i) \sum_{x}(-1)^x \binom(n',xp+j-i)
\equiv
\sum_{i=0}^{p-1}(-1)^i \sum_{x}(-1)^x \binom(n',xp+j-i) \pmod{p^q}\,.
$$
The last (double sum equals $\binom(n',0)-\binom(n',1)+\binom(n',2)-\binom(n',3)+\ldots=0$.
\end{problem}

\begin{problem}
%Докажите, что если $p$ --- нечетное простое, то  $n\kratno p+1$
%тогда и только тогда, когда
%$$
%\binom(n, j)-\binom(n, {j+(p-1)})+
%\binom(n, {j+2(p-1)})-\binom(n, {3(p-1)})+\ldots
%\equiv 0 \pmod{p}
%$$
%при всех $j=1$, 3, \dots, $p-2$.
The result of Bhaskaran (1965), it is cited in \cite{Granville-p^q}, solution [CSTTVZ].

Induction on $n$. Let
$$
f(n,j)=\binom(n, j)-\binom(n, {j+(p-1)})+ \binom(n, {j+2(p-1)})-\binom(n, {j+3(p-1)})+\ldots
$$
The base $n=p+1$ is trivial,
but observe that  $\binom(p+1,i)\equiv 1\pmod{p}$ for $i=0,1,p,{p+1}$,
otherwise this binomial coefficient is divisible by $p$.
Prove the step of induction from $n'=n-(p+1)$ to $n$.
By the observation above we have
\begin{multline*}
\binom({n'+(p+1)}, {j+(p-1)k})=
\sum_{i=0}^{p+1}\binom({n'}, {j+(p-1)k-i})\binom(p+1,i)\equiv
\sum_{i\in\{0,1,p,p+1\}}\binom({n'}, {j+(p-1)k-i})
=\\ =
\binom({n'}, {j+(p-1)k})+\underline{\binom({n'}, {j-1+(p-1)k})+\binom({n'}, {j-1+(p-1)(k-1)})}+
\binom({n'}, {j-2+(p-1)(k-1)}) \pmod{p}\,.
\end{multline*}
Since $f(n,j)=\sum_k (-1)^k\binom(n, {j+k(p-1)})$ is an alternating sum, the underlined summands cancel (except the first and the last,
but these summands are equal to 0 due to incorrect binomial coefficietns). So we obtain the equalities
%\begin{align*}
$$
f(n,j)\equiv f(n',j) - f(n',j-2) \quad\text{ при } j>1\,,
\qquad\qquad\qquad
f(n,1)\equiv f(n',1) + f(n',p-2)\,.
%\end{align*}
$$
Now the part ``only if'' of the problem statement follows from the induction hypothesis,
and the part ``if'', too:
if $f(n,j)\equiv 0 \pmod p$ for $j=1$, 3, \dots, $p-2$, then
$$
f(n',p-2)\equiv f(n',p-4) \equiv \ldots\equiv f(n',1)\equiv -f(n',p-2)\,,
$$
from where $f(n',j)\equiv 0\pmod p$ for all required $j$, and then $n'\kratno (p+1)$, hence $n\kratno (p+1)$.
\end{problem}
