\pitem[]{%
    Show that, for every prime number $p$ and all integer $r$, there is a non negative integer $n$, such that:
    $$\binom{2n}{n}\equiv r \pmod{p}$$
    }{%
    \textbf{Claim 1}. If there exists $n_0$ so $\binom{2n_0}{n_0} \equiv r \pmod{p}$ then there exists $n$ so $\binom{2n}{n} \equiv 2r \pmod{p}$.\\
    \textbf{Proof}. Pick $n=p^k+n_0$ with $p^k>n_0$ then according to Lucas' theorem we have
    $$\binom{2p^k+2n_0}{p^k+n_0} \equiv \binom21 \binom{2n_0}{n_0} \equiv 2r \pmod{p}.$$
    \textbf{Claim 2}. If there exists $n_0$ so that $\binom{2n_0}{n_0} \equiv r \pmod{p}$ with $r<p/2$ then there exists $n$ so $\binom{2n}{n} \equiv 2r-1 \pmod{p}$.\\
    \textbf{Proof}. Note that we have
    \begin{align*} 2^{p-2}\binom{2r-2}{r-1}^{p-2} \binom{2r}{r} & = 2^{p-1} (2r-1)r^{p-2} \left( \frac{(r+1) \cdots (2r-2)}{(r-1)!} \right)^{p-1}, \\ & \equiv (2r-1)r^{p-2} \pmod{p}. \end{align*}Hence, if we pick $n= n_0 \cdot p^k + (\overline{\underbrace{1 \cdots 1}_{p-2}\underbrace{(r-1) \cdots (r-1)}_{p-2} r})_p$ for large enough $k$ then according to Lucas' theorem, we have $$\binom{2n}{n} \equiv \binom{2n_0}{n_0} \binom{2}{1}^{p-2}\binom{2r-2}{r-1}^{p-2}\binom{2r}{r} \equiv r^{p-1}(2r-1) \equiv 2r-1 \pmod{p}.$$From these two claims, by induction, starting with $\binom{2}{1} \equiv 2 \pmod{p}$, we can follow that if there exists $n$ so $\binom{2n}{n} \equiv r \pmod{p}$ for all $r \le k$ then there also exists $n$ so $\binom{2n}{n} \equiv k+1 \pmod{p}$. And $\binom{p+1}{(p+1)/2} \equiv 0 \pmod{p}$. We are done.
