\mysection{Triangle Inequality}

\begin{mysubsection}{}
    \begin{theorem}[thm:]{Triangle Inequality}
        For any triangle with side lengths $a,b\textnormal{ and }c$, where $a\leq b\leq c$

        $a+b>c, a+c>b \textnormal{ and }b+c>a$
    \end{theorem}


\end{mysubsection}

\begin{example}[exp:]{}
    \qitem{%
        The sides of triangles $ABC$ and $ACD$ satisfy the following conditions: $AB = AD = 3$ cm, $BC = 7$ cm, $DC = 11$ cm. What values can the side length $AC$ take if it is an integer number of centimeters, and the largest in $\Delta ABC$?
        }{%
        
        In $ABC$, $7<AC<10$, and in triangle ADC $8<AC<14$.
        So, $AC=9$.

    }
\end{example}
