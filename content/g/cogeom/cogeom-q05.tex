    $P(4,24)$, $Q(q,8)$ and $R(r,8)$ are the vertices of a triangle, where $r>q$. It is given that the area of $\triangle PQR$ is $192$, and $PQ=PR$.
    \begin{enumerate}
        \item Find the values of $q$ and $r$.
        \item Find the equation of the perpendicular bisector of $PR$.
        \item Find the coordinates of circumcentre of $\triangle PQR$.
        \item Find the circumference of the circumcircle of $\triangle PQR$ in terms of $\pi$.
    \end{enumerate}
    }{%
    \begin{enumerate}
        \item $(24-8)(r-q)=192\cdot 2$, $r-q=24$. $16^2+(q-4)^2=16^2+(r-4)^2$, $q=-8$, $r=16$.
        \item $M_{PR}=(10,16)$, $y=\frac{3}{4}x+\frac{17}{2}$.
        \item Another perpendicular bisector: $x=(16-8)/2=4$, Hence $O=(4,23/2)$.
        \item $PO=25/2$, circumference$=25\pi $.
    \end{enumerate}
