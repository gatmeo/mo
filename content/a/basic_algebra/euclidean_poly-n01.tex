\mysection{Euclidean Algorithm for Polynomials}
We can similary perform euclidean algorithms for polynomials using the same setup.
\begin{example}[]{}
    \eitem{%
        Find the greatest common divisor of $x^2+2x+1$ and $x^3+4x^2+6x+3$.
        }{%
        \begin{tabular}{r|cc|l}
           $x+2$  & $x^3+4x^2+6x+3$ & $x^2+2x+1$ & $x+1$ \\
                  &$x^3+4x^2+5x+2$ & $x^2+2x+1$&\\
                  \hline
                  &x+1 & &
        \end{tabular}
    }
\end{example}

\begin{shortque}[Bezout's Identity on Polynomials]{}
    \eitem{%
        Find polynomials $u(x),v(x)$ such that
        \begin{equation*}
            (x^4-1)u(x)+(x^7-1)v(x)=(x-1).
        \end{equation*}
        }{%
        \begin{alignat*}{1}
            x^7-1&= (x^4-1)x^3+x^3-1\\
            x^4-1&= x(x^3-1)+x-1\\
            x^3-1&= (x-1)(x^2+x+1)
        \end{alignat*}
        \begin{alignat*}{1}
            x-1&= x^4-1-x(x^3-1)\\
               &= x^4-1-x(x^7-1-(x^4-1)x^3)\\
               &= (x^4-1)(x^4+1)-x(x^7-1)
        \end{alignat*}
        Hence $u(x)=x^4+1, v(x)=-x$.
    }
\end{shortque}

\begin{shortque}[]{}
    \qitem{%
        Find polynomials $u(x),v(x)$ such that
        \begin{equation*}
            (2x^2-1)u(x)+(3x^3-1)v(x)=1.
        \end{equation*}
        }{%
        \begin{alignat*}{1}
            3x^3-1&= (2x^2-1)(\frac{3}{2}x)+(\frac{3}{2}x-1)\\
            2x^2-1&= (\frac{3}{2}x-1)(\frac{4}{3}x)+(\frac{4}{3}x-1)\\
            2x^2-1&= (\frac{3}{2}x-1)(\frac{4}{3}x+\frac{8}{9})-\frac{1}{9}
        \end{alignat*}

        \begin{alignat*}{1}
            -\frac{1}{9}&= 2x^2-1- (\frac{3}{2}x-1)(\frac{4}{3}x+\frac{8}{9})\\
                        &= 2x^2-1- ((3x^3-1)-(2x^2-1)(\frac{3}{2}x))(\frac{4}{3}x+\frac{8}{9})\\
                        &= (2x^2-1)\left[1-\left(\frac{4}{3}x+\frac{8}{9}\right)\left(-\frac{3}{2}x\right)\right]+(3x^3-1)\left[-\left(\frac{4}{3}x+\frac{8}{9}\right)\right]\\
                        &= (2x^2-1)\left(2x^2+\frac{4}{3}x+1\right)+(3x^3-1)\left[-\left(\frac{4}{3}x+\frac{8}{9}\right)\right].
        \end{alignat*}
        And hence 
        \begin{alignat*}{1}
            1&= (2x^2-1)(-18x^2-12x-9)+(3x^3-1)(12x+8).
        \end{alignat*}
        }{%
        <++>
    }
\end{shortque}
