\qitem{%
    Let $a,b$ be real numbers such that $0\leq a\leq b\leq 1$. 

    Prove that $0\leq ab^2-ba^2\leq \dfrac{1}{4}$.
    }{%
    Note that $b\leq 1$ implies $b^2\leq b$ and hence $-b\leq -b^2$. Hence
    \begin{alignat*}{1}
        ab^2-ba^2&\leq ab^2-b^2a^2=b^2(a-a^2)\\
                 &\leq a-a^2=\frac{1}{4}-(\frac{1}{2}-a^2)\\
                 &\leq \frac{1}{4}
    \end{alignat*}
    }{%
    <++>
}
