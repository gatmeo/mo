\qitem{%
    For $x,y>0$, prove that $\dfrac{1}{x}+\dfrac{1}{y}\geq \dfrac{4}{x+y}$. (Hint: consider $a=\dfrac{1}{x}, b=\dfrac{1}{y}$.)
    }{%
    Apply AM-GM, we get
    \begin{alignat*}{1}
        \frac{1}{x}+\frac{1}{y}&\geq \frac{2}{\sqrt{xy}}\\
        \frac{x+y}{2} &\geq \sqrt{xy}\\
        \frac{1}{\sqrt{xy}}&\geq \frac{2}{x+y}\\
        \frac{1}{x}+\frac{1}{y}&\geq \frac{4}{x+y}
    \end{alignat*}
    \begin{alignat*}{1}
        (x-y)^2&\geq 0\\
        (x+y)^2-4xy&\geq 0\\
        (x+y)^2&\geq 4xy\\
        \frac{x+y}{xy}&\geq \frac{4}{x+y}\\
        \frac{1}{x}+\frac{1}{y}&\geq \frac{4}{x+y}
    \end{alignat*}
    }{%
    <++>
}
