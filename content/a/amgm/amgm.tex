\qitem{%
    For $x,y,z>0$, prove that $(x+y)(y+z)(z+x)\geq 8xyz$.
    }{%
    $x+y\geq 2\sqrt{xy}$
    
    $y+z\geq 2\sqrt{yz}$

    $z+x\geq 2\sqrt{zx}$
    }{%
    <++>
}

\qitem{%
    For $x,y,z>0$, prove that $x^2+y^2+z^2\geq xy+yz+zx$.
    }{%
    $\frac{1}{2} x^2 + \frac{1}{2} y^2\geq xy$

    $\frac{1}{2} y^2 + \frac{1}{2} z^2\geq yz$

    $\frac{1}{2} z^2 + \frac{1}{2} x^2\geq zx$
    }{%
    <++>
}

\qitem{%
    For $x,y,z>0$, prove that $xy+yz+zx\geq x\sqrt{yz}+y\sqrt{zx}+z\sqrt{xy}$.
    }{%
    $\frac{1}{2} xy + \frac{1}{2} yz\geq y\sqrt{xz}$
    }{%
    <++>
}
