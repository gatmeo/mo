\qitem{%
    Let \[A = 2 \cdot 4 \cdot 6 \cdot ... \cdot 7458 + 1 \cdot 3 \cdot 5 \cdot ... \cdot 7457\]Prove that A is divisible by 7459.
    }{%
    \begin{align*}
A&\equiv2\cdot4\cdot6\cdots7458+1\cdot3\cdot5\cdot7457\\
&\equiv2\cdot4\cdot6\cdots7458+(1-7459)(3-7459)(5-7459)\cdots(7457-7459)\\
&\equiv2\cdot4\cdot6\cdots7458+(-7458)(-7456)(-7454)\cdots(-2)\\
&\equiv2\cdot4\cdot6\cdots7458-2\cdot4\cdot6\cdots7458\\
&\equiv0\pmod{7459}
\end{align*}
