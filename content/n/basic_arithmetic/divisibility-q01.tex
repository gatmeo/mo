\qitem{%
    Find the test for divisibility for 19
    }{%
    Add two times the last digit to the remaining leading truncated number.
    \begin{alignat*}{1}
        \overline{abcdefg}&= (19-9)\overline{abcdef} + (19-18)g\\
                          &= 19(\overline{abcdef}) + 19g - (9\overline{abcdef}+18g)\\
                          &= 19(\overline{abcdef}) + 19g - 9(\overline{abcdef}+2g)\\
    \end{alignat*}
    }{%
