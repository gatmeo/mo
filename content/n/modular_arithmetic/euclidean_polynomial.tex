\qitem{%
    What is the sum of all integers $n$ such that $n^2+2n+2$ divides $n^3+4x^2+4x-14$.
    }{%
    $n^3+4n^2+4n-14=(n^2+2n+2)(n+2)+(-2n-18)$

    We need $(n^2+2n+2)|(-2n-18)$. So we have either $-2n-18=0$, or $\left|-2n-18\right|\geq \left|n^2+2n+2\right|$. The first case is $n=-9$, the later we have $-4\leq n\leq 4$, Check every possible answers we have $n=-4,-2,-1,0,1,4$.

    Hence the sum is $-11$.
    }{%
    <++>
}

\qitem{%
    What is the largest positive integer $n$ such that $n^3+100$is divisible by $n+10$?
    }{%
    \begin{alignat*}{1}
        n^3+100&= (n+1)(n^2+an+b) +c\\
               &= n^3+n^2(10+a) + n(b+10a) + 10b + c
    \end{alignat*}

    Solve 
    \begin{equation*}
        \begin{cases}
           &10+a=0\\
           &b+10a=0\\
           &10b+c=100
        \end{cases}
    \end{equation*}

    we have $a=-10,b=100,c=-900$. Hence we have
    $n+10=gcd(n^3,100,n+10)=gcd(-900,n+10)=gcd(900,n+10)$.

    Max value for $n$ is $890$.
    }{%
    <++>
}
