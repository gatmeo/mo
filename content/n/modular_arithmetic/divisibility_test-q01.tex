\qitem{%
    Find and prove the test for divisibility for 29.
    }{%
    Subtract three times the last digit from the remaining leading truncated number.
    \begin{alignat*}{1}
        10a+b&\equiv 10a+30b\equiv 10(a+3b)\\
             &\equiv 29a-19a+30b\equiv -19(a+3b)-87b
    \end{alignat*}
