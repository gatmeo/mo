\qitem{%
    Find polynomials $u(x),v(x)$ such that
    \[(2x^2+3x+3)u(x)+(2x^3+3x^2+5x+1)v(x)=1.\]
    }{%
    \begin{alignat*}{1}
        2&= 2x^2+3x+3 - (x+1)(2x+1)\\
        2x+1&=2x^3+3x^2+5x+1-x(2x^2+3x+3)\\
        2&= (2x^2+3x+3) - (x+1)(2x^3+3x^2+5x+1-x(2x^2+3x+3))\\
         &= (-x-1)(2x^3+3x^2+5x+1) + (1-(x-1)(-x) = x^2-x+1) (2x^2+3x+3)
    \end{alignat*}
    }{%
    <++>
}
