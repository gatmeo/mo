\qitem{%
    (2 pt) Find polynomials $u(x),v(x)$ such that
    \[(x^2-x+3)u(x)+(x^3-x^2+4x-2)v(x)=1.\]
    }{%
    \begin{alignat*}{1}
        x^3-x^2+4x-2&=(x^2-x+3)(x)+(x-2)\\
        x^2-x+3&=(x-2)(x+1)+(5)\\
        x-2&=(5)(x/5-2/5)+(0)
    \end{alignat*}

    $gcd(x3-x2+4x-2,x2-x+3)=1$
    $(x^2-x+3)(\frac{x^2}{5}+\frac{x}{5}+1/5)+(x^3-x^2+4x-2)(\frac{-x}{5}+\frac{-1}{5})=1$
    }{%
