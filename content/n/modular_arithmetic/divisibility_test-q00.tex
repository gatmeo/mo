\qitem{%
    Find and prove the test for divisibility for 31.
    }{%
    Subtract three times the last digit from the remaining leading truncated number.
    \begin{alignat*}{1}
        10a+b&\equiv 10a-30b\equiv 10(a-3b)\\
             &\equiv 31a-21a-30b\equiv -21(a-3b)-93b
    \end{alignat*}
